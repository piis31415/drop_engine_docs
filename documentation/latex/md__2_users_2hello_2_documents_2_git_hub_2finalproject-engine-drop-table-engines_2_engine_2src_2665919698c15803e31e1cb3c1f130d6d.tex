\begin{DoxyItemize}
\item {\ttfamily \{ owner\+: \{ uid\+: 100, gid\+: 100 \} \}} -\/ set the {\ttfamily uid} and {\ttfamily gid} explicitly
\item {\ttfamily \{ owner\+: 100 \}} -\/ use one value, will set both {\ttfamily uid} and {\ttfamily gid} the same
\end{DoxyItemize}
